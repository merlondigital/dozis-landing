    1. **Create `middleware.ts`** at project root — Next.js middleware for route protection:
       ```typescript
       import { NextResponse } from "next/server";
       import type { NextRequest } from "next/server";

       export function middleware(request: NextRequest) {
         const { pathname } = request.nextUrl;

         // Check for better-auth session cookie
         // better-auth uses "better-auth.session_token" cookie name by default
         const sessionToken = request.cookies.get("better-auth.session_token");

         // Public routes that don't need auth
         if (pathname === "/" || pathname.startsWith("/api/") || pathname === "/app/login") {
           return NextResponse.next();
         }

         // All /app/* routes require authentication
         if (pathname.startsWith("/app")) {
           if (!sessionToken) {
             const loginUrl = new URL("/app/login", request.url);
             // Store original URL for redirect after login (per D-18)
             loginUrl.searchParams.set("callbackUrl", pathname);
             return NextResponse.redirect(loginUrl);
           }
         }

         return NextResponse.next();
       }

       export const config = {
         matcher: [
           // Match all /app/* routes except static files
           "/app/:path*",
         ],
       };
       ```

       **NOTE:** The middleware does a lightweight cookie check only (not full session validation — that happens server-side in layouts). This is the standard Next.js pattern: middleware for fast redirects, server components for full validation.

       **NOTE:** The exact cookie name used by better-auth must be verified. Check better-auth docs for the default session cookie name. It may be `better-auth.session_token` or `better-auth.session` or something else.
