    4. **Create `src/lib/auth.ts`** — better-auth server with email-otp plugin and AUTH-02 rate limiting:
       ```typescript
       import { betterAuth } from "better-auth";
       import { emailOTP } from "better-auth/plugins";
       import { drizzleAdapter } from "better-auth/adapters/drizzle";
       import { getDb } from "@/src/db";
       import { sendOtpEmail } from "./email";
       import type { CloudflareEnv } from "@/src/env";

       export function createAuth(env: CloudflareEnv) {
         const db = getDb(env.DB);

         return betterAuth({
           database: drizzleAdapter(db, {
             provider: "sqlite",
           }),
           secret: env.BETTER_AUTH_SECRET,
           baseURL: env.BETTER_AUTH_URL || "http://localhost:3000",
           emailAndPassword: {
             enabled: false, // OTP only, no passwords
           },
           session: {
             expiresIn: 60 * 60 * 24 * 30, // 30 days per D-16
             cookieCache: {
               enabled: true,
               maxAge: 60 * 5, // 5 min cache
             },
           },
           rateLimit: {
             window: 60 * 15, // 15-minute window per AUTH-02
             max: 5, // max 5 attempts per email per 15 minutes per AUTH-02 / D-17
           },
           plugins: [
             emailOTP({
               otpLength: 6,
               expiresIn: 300, // 5 minutes per D-17
               sendVerificationOTP: async ({ email, otp, type }) => {
                 await sendOtpEmail({
                   email,
                   otp,
                   resendApiKey: env.RESEND_API_KEY,
                 });
               },
             }),
           ],
           user: {
             additionalFields: {
               role: {
                 type: "string",
                 required: false,
                 defaultValue: "user",
                 input: false, // cannot be set by client
               },
               lastName: {
                 type: "string",
                 required: false,
               },
               firstName: {
                 type: "string",
                 required: false,
               },
               birthYear: {
                 type: "number",
                 required: false,
               },
               address: {
                 type: "string",
                 required: false,
               },
               profileCompleted: {
                 type: "boolean",
                 required: false,
                 defaultValue: false,
                 input: false,

             },

         });
       ```

    2. **Create `src/lib/auth-utils.ts`** — server-side session utilities:
       ```typescript
       import { createAuth } from "./auth";
       import { headers } from "next/headers";
       import { getCloudflareContext } from "@opennextjs/cloudflare";
       import type { CloudflareEnv } from "@/src/env";

       /**
        * Get the current session on the server side.
        * Use in Server Components and Server Actions.
        */
       export async function getSession() {
         const { env } = await getCloudflareContext();
         const auth = createAuth(env as unknown as CloudflareEnv);
         const session = await auth.api.getSession({
           headers: await headers(),
         });
         return session;
       }

       /**
        * Check if the current user is an admin.
        * Returns the session if admin, null otherwise.
        */
       export async function requireAdmin() {
         const session = await getSession();
         if (!session?.user || session.user.role !== "admin") {
           return null;
         }
         return session;
       }

       /**
        * Check if the current user has completed their profile.
        */
       export async function requireProfile() {
         const session = await getSession();
         if (!session?.user) return null;
         if (!session.user.profileCompleted) return { needsProfile: true, session };
         return { needsProfile: false, session };
       }

       /**
        * Mask email for display: m***n@gmail.com
        * Per D-15: Email partially masked on OTP step.
        */
       export function maskEmail(email: string): string {
         const [local, domain] = email.split("@");
         if (local.length <= 2) return `${local[0]}***@${domain}`;
         return `${local[0]}***${local[local.length - 1]}@${domain}`;
       }
       ```

       **NOTE:** Verify `auth.api.getSession` usage pattern against better-auth docs. The pattern of passing `headers` from Next.js `headers()` function must work with better-auth's session resolution.
